上游调用 notification 的方式

  方式一：HTTP 调用

  直接调用 notification 暴露的 HTTP 接口：

  # 邮件发送
  curl -X POST "http://<notification-host>:8080/api/apex/notification/v1/mail/notification" \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer <token>" \
    -d '{
      "subject": "告警通知",
      "text": "告警内容",
      "to": ["user@example.com"]
    }'
  # 短信发送
  curl -X POST "http://<notification-host>:8080/api/apex/notification/v1/sms/notification" \
    -H "Content-Type: application/json" \
    -d '{
      "templateKey": "commonTemplate",
      "recipients": ["13111223344"],
      "platform": "ali",
      "templateParams": [{"key": "code", "value": "123456"}]
    }'

  # 企业微信
  curl -X POST "http://<notification-host>:8080/api/apex/notification/v1/weCom/notification" \
    -H "Content-Type: application/json" \
    -d '{
      "secret": "xxx",
      "agentId": "1000108",
      "phone": ["13111223344"],
      "text": "测试消息"
    }'
